\begin{verbatim}
function [y]=taodayso(a,b,n)
 if nargin<3
   n=10;
 end
 y=zeros(1,n); y(1)=a; y(2)=b;
 k=3;
 while k<=n
   y(k)=y(k-1)+y(k-2);
   k=k+1;
 end 
\end{verbatim}
